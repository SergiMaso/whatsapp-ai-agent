    def save_customer_language(self, phone, language):
        """
        Guardar l'idioma preferit d'un client
        
        IMPORTANT: Nom√©s guarda l'idioma, NO crea client amb nom "Usuario"
        """
        try:
            conn = self.get_connection()
            cursor = conn.cursor()
            
            # Verificar si el client ja existeix
            cursor.execute("SELECT name FROM customers WHERE phone = %s", (phone,))
            existing = cursor.fetchone()
            
            if existing:
                # Client existent: nom√©s actualitzar idioma
                cursor.execute("""
                    UPDATE customers
                    SET language = %s, last_visit = CURRENT_TIMESTAMP
                    WHERE phone = %s
                """, (language, phone))
            else:
                # Client nou: nom√©s guardar idioma sense nom
                # Quan faci la reserva, es guardar√† el nom real
                cursor.execute("""
                    INSERT INTO customers (phone, name, language, last_visit)
                    VALUES (%s, '', %s, CURRENT_TIMESTAMP)
                """, (phone, language))
            
            conn.commit()
            cursor.close()
            conn.close()
            print(f"üåç Idioma guardado: {phone} ‚Üí {language}")
        except Exception as e:
            print(f"‚ùå Error guardando idioma: {e}")
